programa
inteiro num1, num2, soma;
decimal media;
{
    escreva("Digite o primeiro número: ");
    leia(num1);
    escreva("Digite o segundo número: ");
    leia(num2);
    soma := num1 + num2;
    media := soma / 2;
    escreva("A média é: ");
    escreva(media);
}
fimprog;
